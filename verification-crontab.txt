# Notes:

# Configure your install location (REQUIRED)
AVS_HOME=/path/to/agent-verification-system

# Optional: integrate with an external workspace that contains autonomous-queue.yaml
# AVS_WORKSPACE=/path/to/your/workspace

# Tier 0: Queue Executor (every 20 minutes)
*/20 * * * * cd $AVS_HOME && bash bin/queue-executor.sh >> /tmp/avs-executor-cron.log 2>&1

# Tier 2: Task Verifier (every 10 minutes)
*/10 * * * * cd $AVS_HOME && bash bin/verify-recent-tasks.sh >> /tmp/avs-verifier-cron.log 2>&1

# Queue Sync (every 15 minutes)
*/15 * * * * cd $AVS_HOME && python3 lib/queue-sync-artifacts.py >> /tmp/avs-queue-sync-cron.log 2>&1

# Tier 3: Meta-Monitor (every 30 minutes)
*/30 * * * * cd $AVS_HOME && bash bin/check-system-health.sh >> /tmp/avs-meta-monitor-cron.log 2>&1
